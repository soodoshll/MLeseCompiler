int ans;
int n;
int get(int x) {
	if(x == ans || x == 1) {
		return ans;
	}
	if((x & 1) == 1) {
		return get(x - 1);
	}
	else {
		return get(x - 1);
	}
	return 1;
}
int main()
{
	n = getInt();
	ans = 1;
	int i = 1;
	for(i = 1; i <= n; i++)
	{
		ans = (ans & get(i));
		//println(toString(i));
	}
	return ans;
}
